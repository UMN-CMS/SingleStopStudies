\def\gaussianTwoD#1#2#3{
  \pgfmathdeclarefunction{#1}{2}{
    \pgfmathparse{exp( -((##1 - #2)^2 + (##2-#3)^2 ))}
  }
}

\def\gaussianOneD#1#2{
  \pgfmathdeclarefunction{#1}{1}{\pgfmathparse{2.71^(-((##1 - #2)^2))}}
}
\gaussianOneD{mygaussian}{0}
